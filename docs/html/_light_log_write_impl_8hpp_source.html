246 void WriteLogContent(const std::wstring& sTypeVal, const std::wstring& sMessage) {
247 bool bNeedReport = false;
248 size_t currentDiscard = 0;
249 static thread_local bool inErrorReport = false;
250
251 if (queueFullStrategy == LogQueueOverflowStrategy::Block) {
252 std::unique_lock<std::mutex> sWriteLock(pLogWriteMutex);
253 // std::wcerr << L"[WriteLogContent] Try push (Block), queue size: " <<
254 // pLogWriteQueue.size() << std::endl;
255 pWrittenCondVar.wait(sWriteLock, [this] { return pLogWriteQueue.size() < kMaxQueueSize; });
256 pLogWriteQueue.push({ sTypeVal, sMessage });
257 // std::wcerr << L"[WriteLogContent] Pushed (Block), queue size: " <<
258 // pLogWriteQueue.size() << std::endl;
259 }
260 else if (queueFullStrategy == LogQueueOverflowStrategy::DropOldest) {
261 std::lock_guard<std::mutex> sWriteLock(pLogWriteMutex);
262 // std::wcerr << L"[WriteLogContent] Try push (DropOldest), queue size: " <<
263 // pLogWriteQueue.size() << std::endl;
264 if (pLogWriteQueue.size() >= kMaxQueueSize) {
265 // std::wcerr << L"[WriteLogContent] Drop oldest, queue full: " <<
266 // pLogWriteQueue.size() << std::endl;
267 pLogWriteQueue.pop();
268 ++discardCount;
269 if (discardCount - lastReportedDiscardCount >= reportInterval) {
270 bNeedReport = true;
271 currentDiscard = discardCount;
272 lastReportedDiscardCount.store(discardCount.load());
273 }
274 }
275 pLogWriteQueue.push({ sTypeVal, sMessage });
276 // std::wcout << L"[WriteLogContent] Pushed (DropOldest), queue size: " <<
277 // pLogWriteQueue.size() << std::endl;
278 }
279 pWrittenCondVar.notify_one();
280 if (bNeedReport && !inErrorReport) {
281 inErrorReport = true;
282 std::wstring overflowMsg = L"The log queue overflows and has been discarded " + std::to_wstring(currentDiscard) + L" logs";
283 // std::wcerr << L"[WriteLogContent] Report overflow: " << overflowMsg << std::endl;
284 WriteLogContent(L"LOG_OVERFLOW", overflowMsg);
285 inErrorReport = false;
286 }
287 }

375 void RunWriteThread() {
376 while (true) {
377 if (bHasLogLasting)
378 if (bLastingTmTags != (GetCurrsTimerTm().tm_hour > 12))
379 CreateLogsFile();
380 LightLogWriteInfo sLogMessageInf;
381
382 {
383 auto sLock = std::unique_lock<std::mutex>(pLogWriteMutex);
384 pWrittenCondVar.wait(sLock, [this]
385 { return !pLogWriteQueue.empty() || bIsStopLogging; });
386
387 if (bIsStopLogging && pLogWriteQueue.empty())
388 break; // 如果停止标志为真且队列为空，则退出线程
389 if (!pLogWriteQueue.empty()) {
390 sLogMessageInf = pLogWriteQueue.front();
391 pLogWriteQueue.pop();
392 pWrittenCondVar.notify_one();
393 }
394 }
395 if (!sLogMessageInf.sLogContentVal.empty() && pLogFileStream.is_open()) {
396 pLogFileStream << sLogMessageInf.sLogTagNameVal << L"-//>>>" << GetCurrentTimer() << " : " << sLogMessageInf.sLogContentVal << "\n";
397 }
398 }
399 pLogFileStream.close();
400 std::cerr << "Log write thread Exit\n";
401 }
