125template <typename T>
126class LockFreeQueue {
127public:
128 explicit LockFreeQueue(size_t capacity)
129 {
130 _capacityMask = capacity - 1;
131 for (size_t i = 1; i <= sizeof(void*) * 4; i <<= 1)
132 _capacityMask |= _capacityMask >> i;
133 _capacity = _capacityMask + 1;
134
135 _queue = (Node*)new char[sizeof(Node) * _capacity];
136 for (size_t i = 0; i < _capacity; ++i)
137 {
138 _queue[i].tail.store(i, std::memory_order_relaxed);
139 _queue[i].head.store(-1, std::memory_order_relaxed);
140 }
141
142 _tail.store(0, std::memory_order_relaxed);
143 _head.store(0, std::memory_order_relaxed);
144 }
145
146 ~LockFreeQueue()
147 {
148 for (size_t i = _head; i != _tail; ++i)
149 (&_queue[i & _capacityMask].data)->~T();
150
151 delete[](char*)_queue;
152 }
153
154 size_t capacity() const { return _capacity; }
155
156 size_t size() const
157 {
158 size_t head = _head.load(std::memory_order_acquire);
159 return _tail.load(std::memory_order_relaxed) - head;
160 }
161
162 bool push(const T& data)
163 {
164 Node* node;
165 size_t tail = _tail.load(std::memory_order_relaxed);
166 for (;;)
167 {
168 node = &_queue[tail & _capacityMask];
169 if (node->tail.load(std::memory_order_relaxed) != tail)
170 return false;
171 if ((_tail.compare_exchange_weak(tail, tail + 1, std::memory_order_relaxed)))
172 break;
173 }
174 new (&node->data)T(data);
175 node->head.store(tail, std::memory_order_release);
176 return true;
177 }
178
179 bool pop(T& result)
180 {
181 Node* node;
182 size_t head = _head.load(std::memory_order_relaxed);
183 for (;;)
184 {
185 node = &_queue[head & _capacityMask];
186 if (node->head.load(std::memory_order_relaxed) != head)
187 return false;
188 if (_head.compare_exchange_weak(head, head + 1, std::memory_order_relaxed))
189 break;
190 }
191 result = node->data;
192 (&node->data)->~T();
193 node->tail.store(head + _capacity, std::memory_order_release);
194 return true;
195 }
196
197private:
198 struct Node
199 {
200 T data;
201 std::atomic<size_t> tail;
202 std::atomic<size_t> head;
203 };

205private:

207 Node* _queue;
208 size_t _capacity;
209 char cacheLinePad1[64];
210 std::atomic<size_t> _tail;
211 char cacheLinePad2[64];
212 std::atomic<size_t> _head;
213 char cacheLinePad3[64];

271 void WriteLogContent(const std::wstring& sTypeVal, const std::wstring& sMessage) {
272 bool bNeedReport = false;
273 size_t currentDiscard = 0;
274 static thread_local bool inErrorReport = false;
275
276 if (queueFullStrategy == LogQueueOverflowStrategy::Block) {
277 // 阻塞直到成功写入
278 while (!pLogWriteQueue.push({ sTypeVal, sMessage })) {
279 if (bIsStopLogging) return;
280 std::this_thread::yield();
281 }
282 }
283 else if (queueFullStrategy == LogQueueOverflowStrategy::DropOldest) {
284 if (!pLogWriteQueue.push({ sTypeVal, sMessage })) {
285 // 队列满，丢弃最老的再插入
286 LightLogWrite_Info dummy;
287 pLogWriteQueue.pop(dummy);
288 pLogWriteQueue.push({ sTypeVal, sMessage });
289 ++discardCount;
290 if (discardCount - lastReportedDiscardCount >= reportInterval) {
291 bNeedReport = true;
292 currentDiscard = discardCount;
293 lastReportedDiscardCount.store(discardCount.load());
294 }
295 }
296 }
297 pWrittenCondVar.notify_one();
298
299 if (bNeedReport && !inErrorReport) {
300 inErrorReport = true;
301 std::wstring overflowMsg = L"The log queue overflows and has been discarded "
302 + std::to_wstring(currentDiscard) + L" logs";
303 //std::wcerr << L"[WriteLogContent] Report overflow: " << overflowMsg << std::endl;
304 WriteLogContent(L"LOG_OVERFLOW", overflowMsg);
305 inErrorReport = false;
306 }
307 }

351 void RunWriteThread() {
352 while (true) {
353 // 检查是否需要切换日志文件（AM/PM切换）
354 if (bHasLogLasting) {
355 bool isCurrentPM = (GetCurrsTimerTm().tm_hour > 12);
356 if (bLastingTmTags != isCurrentPM) {
357 CreateLogsFile();
358 }
359 }
360
361 LightLogWrite_Info sLogMessageInf;
362 bool hasLog = pLogWriteQueue.pop(sLogMessageInf);
363
364 // 只有在停止标志为真且队列为空时才退出
365 if (bIsStopLogging && pLogWriteQueue.size() == 0 && !hasLog) {
366 break;
367 }
368
369 // 写入日志内容到文件
370 if (hasLog && !sLogMessageInf.sLogContentVal.empty() && pLogFileStream.is_open()) {
371 pLogFileStream << sLogMessageInf.sLogTagNameVal
372 << L"-//>>>" << GetCurrentTimer()
373 << L" : " << sLogMessageInf.sLogContentVal << L"\n";
374 }
375 else if (!hasLog) {
376 // 队列为空，避免忙等
377 std::this_thread::sleep_for(std::chrono::milliseconds(10));
378 }
379 }
380
381 // 关闭日志文件流
382 pLogFileStream.close();
383 std::cerr << "Log write thread Exit\n";
384 }
